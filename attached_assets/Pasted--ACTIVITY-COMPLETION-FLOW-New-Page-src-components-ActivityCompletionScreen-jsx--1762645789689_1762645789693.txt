### Endpoint: `/api/activity-of-day`

```javascript
// backend/routes/activityOfDay.js
const router = express.Router();

router.post('/activity-of-day', async (req, res) => {
  const { baby_name, age_months, development_goals } = req.body;

  try {
    // Check if activity already exists for today
    const existing = await supabase
      .from('activity_of_day')
      .select('*')
      .eq('date', new Date().toISOString().split('T')[0])
      .eq('baby_id', baby_id)
      .single();

    if (existing.data) {
      return res.json(existing.data);
    }

    // Generate new activity via LLM
    const activity = await generateActivityViaLLM(
      baby_name,
      age_months,
      development_goals
    );

    // Save to Supabase
    await supabase.from('activity_of_day').insert({
      baby_id,
      activity_json: activity,
      date: new Date().toISOString().split('T')[0],
      created_at: new Date().toISOString(),
    });

    res.json(activity);
  } catch (error) {
    console.error('Error fetching activity of day:', error);
    res.status(500).json({ error: 'Failed to generate activity' });
  }
});

async function generateActivityViaLLM(babyName, ageMonths, goals) {
  const systemPrompt = `You are an empathetic early-childhood development coach designing joyful bonding activities for Nurtura.
Generate ONE Activity of the Day for a parent and child. Output ONLY valid JSON, no markdown.`;

  const userPrompt = `
Child: ${babyName}, ${ageMonths} months old
Parent goals: ${goals.join(', ')}

Generate a â‰¤10 minute activity using only household items. Output JSON:
{
  "title": "Activity name",
  "age_range": "Xâ€“Y months or years",
  "duration": "X min",
  "domain": "Developmental area(s)",
  "goal_focus": "Focus area",
  "short_teaser": "One-line engaging teaser",
  "what_you_need": "Simple materials list",
  "steps": ["Step 1", "Step 2", ...],
  "why_it_matters": "Developmental benefit",
  "end_moment": "How to end joyfully",
  "emotional_tone": "Tone (e.g., Playful, Calm)",
  "engagement_hint": "E.g., '1,230 parents played this today ğŸ’–'"
}`;

  const response = await openai.createChatCompletion({
    model: 'gpt-4',
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: userPrompt },
    ],
    temperature: 0.7,
    max_tokens: 800,
  });

  const activityJson = JSON.parse(response.choices[0].message.content);
  return activityJson;
}

router.post('/activity-completed', async (req, res) => {
  const { activity_id, completed_at } = req.body;

  try {
    await supabase.from('activity_completions').insert({
      activity_id,
      completed_at,
    });

    res.json({ success: true });
  } catch (error) {
    res.status(500).json({ error: 'Failed to mark activity as complete' });
  }
});
```
