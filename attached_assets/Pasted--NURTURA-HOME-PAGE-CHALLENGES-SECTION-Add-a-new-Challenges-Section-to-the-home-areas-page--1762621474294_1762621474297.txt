```sql
-- Store challenge templates
CREATE TABLE challenges (
  id INTEGER PRIMARY KEY,
  duration_days INTEGER, -- 30, 90, 180, 365
  title TEXT,
  tagline TEXT, -- e.g., "Build Curiosity & Wonder"
  description TEXT, -- Lucrative, emotional description
  cover_image TEXT, -- emoji or color
  development_types TEXT, -- JSON: ["Physical", "Cognitive"]
  created_at TIMESTAMP
);

-- Store daily activities for each challenge
CREATE TABLE challenge_activities (
  id INTEGER PRIMARY KEY,
  challenge_id INTEGER,
  day_number INTEGER, -- 1, 2, 3... 365
  activity_title TEXT,
  activity_description TEXT,
  materials TEXT, -- JSON array
  how_to TEXT, -- JSON array of steps
  why_it_helps TEXT,
  duration_min INTEGER,
  created_at TIMESTAMP,
  FOREIGN KEY (challenge_id) REFERENCES challenges(id)
);

-- Track parent enrollment in challenges
CREATE TABLE challenge_enrollments (
  id INTEGER PRIMARY KEY,
  baby_id INTEGER,
  challenge_id INTEGER,
  started_at TIMESTAMP,
  completed_days INTEGER DEFAULT 0,
  status TEXT, -- "active", "paused", "completed"
  created_at TIMESTAMP,
  FOREIGN KEY (baby_id) REFERENCES babies(id),
  FOREIGN KEY (challenge_id) REFERENCES challenges(id)
);

-- Track daily completion
CREATE TABLE challenge_daily_logs (
  id INTEGER PRIMARY KEY,
  enrollment_id INTEGER,
  day_number INTEGER,
  activity_id INTEGER,
  completed_at TIMESTAMP,
  reflection_notes TEXT, -- Parent notes
  created_at TIMESTAMP,
  FOREIGN KEY (enrollment_id) REFERENCES challenge_enrollments(id),
  FOREIGN KEY (activity_id) REFERENCES challenge_activities(id)
);
```

@app.route('/home')
@login_required
def home():
    """
    Show home/areas page with new Challenges section
    """
    user_id = session['user_id']
    baby = get_baby_by_user(user_id)
    
    if not baby:
        return redirect('/onboarding')
    
    # Get areas
    areas = db.execute(
        'SELECT * FROM development_areas WHERE baby_id = ? ORDER BY development_type',
        (baby.id,)
    ).fetchall()
    
    if not areas:
        areas = []
    
    # Get challenges (all 4 templates)
    challenges = db.execute(
        'SELECT * FROM challenges ORDER BY duration_days'
    ).fetchall()
    
    if not challenges:
        # First visit: Generate challenge templates
        challenges = generate_challenge_templates()
        
        for challenge in challenges:
            db.execute(
                '''INSERT INTO challenges 
                   (duration_days, title, tagline, description, cover_image, development_types)
                   VALUES (?, ?, ?, ?, ?, ?)''',
                (challenge['duration'], challenge['title'], challenge['tagline'], 
                 challenge['description'], challenge['emoji'], 
                 json.dumps(challenge['development_types']))
            )
        db.commit()
        challenges = db.execute(
            'SELECT * FROM challenges ORDER BY duration_days'
        ).fetchall()
    
    # Get parent's active challenges
    active_challenges = db.execute(
        '''SELECT ce.*, c.title, c.duration_days, c.tagline 
           FROM challenge_enrollments ce
           JOIN challenges c ON ce.challenge_id = c.id
           WHERE ce.baby_id = ? AND ce.status = 'active'
           ORDER BY ce.started_at DESC''',
        (baby.id,)
    ).fetchall()
    
    return render_template('home.html', 
                         baby=baby,
                         areas=areas,
                         challenges=challenges,
                         active_challenges=active_challenges)

@app.route('/api/challenge/<int:challenge_id>')
def get_challenge_details(challenge_id):
    """
    Get challenge details with all daily activities
    """
    challenge = db.execute(
        'SELECT * FROM challenges WHERE id = ?',
        (challenge_id,)
    ).fetchone()
    
    if not challenge:
        return jsonify({'error': 'Challenge not found'}), 404
    
    activities = db.execute(
        'SELECT * FROM challenge_activities WHERE challenge_id = ? ORDER BY day_number',
        (challenge_id,)
    ).fetchall()
    
    return jsonify({
        'id': challenge['id'],
        'title': challenge['title'],
        'tagline': challenge['tagline'],
        'description': challenge['description'],
        'duration_days': challenge['duration_days'],
        'activities': [
            {
                'day': a['day_number'],
                'title': a['activity_title'],
                'description': a['activity_description']
            }
            for a in activities
        ]
    })

@app.route('/api/generate-challenge-activities/<int:challenge_id>')
def generate_challenge_activities(challenge_id):
    """
    Generate daily activities for a challenge using Claude
    """
    challenge = db.execute(
        'SELECT * FROM challenges WHERE id = ?',
        (challenge_id,)
    ).fetchone()
    
    if not challenge:
        return jsonify({'error': 'Challenge not found'}), 404
    
    # Check if activities already exist
    existing = db.execute(
        'SELECT COUNT(*) as count FROM challenge_activities WHERE challenge_id = ?',
        (challenge_id,)
    ).fetchone()
    
    if existing['count'] > 0:
        return jsonify({'message': 'Activities already generated'})
    
    # Generate activities
    activities = generate_daily_activities(challenge)
    
    # Store in DB
    for activity in activities:
        db.execute(
            '''INSERT INTO challenge_activities 
               (challenge_id, day_number, activity_title, activity_description, 
                materials, how_to, why_it_helps, duration_min)
               VALUES (?, ?, ?, ?, ?, ?, ?, ?)''',
            (challenge_id, activity['day'], activity['title'], activity['description'],
             json.dumps(activity.get('materials', [])), json.dumps(activity.get('how_to', [])),
             activity.get('why_it_helps', ''), activity.get('duration', 10))
        )
    db.commit()
    
    return jsonify({'success': True, 'activities_count': len(activities)})


def generate_daily_activities(challenge):
    """
    Generate continuous daily activities for a challenge using Claude
    """
    duration = challenge['duration_days']
    title = challenge['title']
    tagline = challenge['tagline']
    
    prompt = f"""
    You are a child development expert. Generate {duration} continuous daily activities for a parent-child bonding challenge.
    
    Challenge: {title}
    Duration: {duration} days
    Focus: {tagline}
    
    IMPORTANT: These are CONTINUOUS activities, meaning:
    - Activities build on each other progressively
    - Each day introduces something new but relates to previous days
    - Skills compound over time (e.g., Day 1: listen to music, Day 2: sing together, Day 3: make instruments, Day 4: create songs)
    - Create progression in difficulty, complexity, and independence
    - By day {duration}, significant habit/skill development should be visible
    
    Format as JSON:
    {{
      "activities": [
        {{
          "day": 1,
          "title": "Start Listening",
          "description": "Introduce your child to different sounds. Play soft music, nature sounds, or nursery rhymes.",
          "materials": ["Phone or speaker", "Headphones (optional)"],
          "how_to": [
            "Find a quiet space",
            "Play 5-10 minutes of gentle music or nature sounds",
            "Observe your child's reactions",
            "Talk about what they hear"
          ],
          "why_it_helps": "Builds auditory awareness and sensory development.",
          "duration": 10
        }},
        {{
          "day": 2,
          "title": "Sound Exploration",
          "description": "Expand sound discovery by creating simple sounds together.",
          "materials": ["Spoons", "Pots", "Bells"],
          "how_to": [
            "Collect safe household items that make sounds",
            "Make different sounds together",
            "Let child bang and explore",
            "Name each sound (high, low, loud, soft)"
          ],
          "why_it_helps": "Develops cause-effect understanding and language.",
          "duration": 10
        }},
        ... ({duration} activities total)
      ]
    }}
    
    Guidelines:
    1. Progression: Each day builds on previous knowledge/skills
    2. Variety: Mix different activity types (active, quiet, learning, creative)
    3. Age-Agnostic: Works for ages 0-6 (adjust complexity in description)
    4. Accessible: Use household items, no expensive materials
    5. Realistic: Each activity takes 5-15 minutes
    6. Engaging: Make descriptions exciting and parent-focused
    7. Habit-Building: By day {duration}, a visible habit or skill is formed
    
    Return ONLY valid JSON, no markdown.
    """
    
    response = client.messages.create(
        model="claude-3-5-sonnet-20241022",
        max_tokens=8000 if duration == 30 else 15000,
        messages=[{"role": "user", "content": prompt}]
    )
    
    data = json.loads(response.content[0].text)
    return data['activities']
